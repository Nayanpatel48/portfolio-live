NumPy Basics

Task Definition: Learned NumPy arrays, array operations, reshaping, and mathematical computations for numerical data. 
import numpy as np

# Creating Arrays
arr = np.array([1, 2, 3, 4, 5])
matrix = np.array([[1, 2], [3, 4]])

# Array Operations
result = arr * 2  # [2, 4, 6, 8, 10]
sum_arr = np.sum(arr)

# Reshaping
reshaped = arr.reshape(5, 1)

# Mathematical Operations
mean_val = np.mean(arr)
std_val = np.std(arr)
