Data Cleaning

Task Definition: Handled missing values, duplicates, and data type conversions using Pandas. 
import pandas as pd

# Handling Missing Values
df.dropna()  # Remove rows with NaN
df.fillna(0)  # Fill NaN with 0
df.fillna(df.mean())  # Fill with mean

# Removing Duplicates
df.drop_duplicates()

# Data Type Conversion
df['column'] = df['column'].astype('int')
df['date'] = pd.to_datetime(df['date'])

Data Visualization

Task Definition: Created line plots, bar charts, histograms, and scatter plots using Matplotlib. 
import matplotlib.pyplot as plt
import pandas as pd

# Line Plot
plt.plot(df['x'], df['y'])
plt.title('Line Plot')
plt.show()

# Bar Chart
df['category'].value_counts().plot(kind='bar')
plt.title('Bar Chart')
plt.show()

# Histogram
df['values'].hist(bins=20)
plt.title('Histogram')
plt.show()

# Scatter Plot
plt.scatter(df['x'], df['y'])
plt.title('Scatter Plot')
plt.show()
